{
    alias wchar wint_t;
    alias wchar wchar_t;
    alias wchar wctype_t;
    alias wchar wctrans_t;

    const wchar WEOF = 0xFFFF;

    alias dchar wchar_t;
    alias dchar wctype_t;
    alias dchar wctrans_t;

    const dchar WEOF = 0xFFFF;
}
